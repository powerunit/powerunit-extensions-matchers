20  package ch.powerunit.extensions.matchers.provideprocessor.extension.hamcrestdate;
21  
22  import java.util.ArrayList;
23  import java.util.Arrays;
24  import java.util.Collection;
25  import java.util.Collections;
26  import java.util.function.Supplier;
27  import java.util.stream.Collectors;
28  
29  import javax.lang.model.type.TypeMirror;
30  
31  import ch.powerunit.extensions.matchers.provideprocessor.ProvidesMatchersAnnotatedElementData;
32  import ch.powerunit.extensions.matchers.provideprocessor.RoundMirror;
33  import ch.powerunit.extensions.matchers.provideprocessor.dsl.DSLMethod;
34  import ch.powerunit.extensions.matchers.provideprocessor.extension.AutomatedExtension;
35  import ch.powerunit.extensions.matchers.provideprocessor.fields.AbstractFieldDescription;
36  import ch.powerunit.extensions.matchers.provideprocessor.fields.FieldDSLMethod;
37  
38  /**
39   * @author borettim
40   *
41   */
42  public abstract class AbstractHamcrestDateMatchersAutomatedExtension extends AutomatedExtension {
43  
44  	private final TypeMirror knownType;
45  
46  	private final String targetType;
47  
48  	private final boolean withSameDay;
49  
50  	public AbstractHamcrestDateMatchersAutomatedExtension(RoundMirror roundMirror, String targetElement,
51  			String targetType, boolean withSameDay) {
52  		super(roundMirror, targetElement);
53  		this.targetType = targetType;
54  		knownType = getMirrorFor(targetType);
55  		this.withSameDay = withSameDay;
56  	}
57  
58  	private FieldDSLMethod builderFor(AbstractFieldDescription field, String targetMethod) {
59  		String suffix = targetMethod.substring(0, 1).toUpperCase() + targetMethod.substring(1);
60  		String readable = Arrays.stream(targetMethod.split("(?=[A-Z])")).collect(Collectors.joining(" ")).toLowerCase();
61  		String target = getExpectedElement();
62  		return builderFor(field).withDeclaration(suffix, targetType + " date")
63  				.withJavaDoc("Verify that this `" + targetType + "` is " + readable + " another one",
64  						"date the `" + targetType + "` to compare with",
65  						target + "#" + targetMethod + "(" + targetType + ")")
66  				.havingDefault(target + "." + targetMethod + "(date)");
67  	}
68  
69  	/*
70  	 * (non-Javadoc)
71  	 * 
72  	 * @see ch.powerunit.extensions.matchers.provideprocessor.extension.
73  	 * AutomatedExtension#accept(ch.powerunit.extensions.matchers.
74  	 * provideprocessor.fields.FieldDescriptionMetaData)
75  	 */
76  	@Override
77  	public Collection<FieldDSLMethod> accept(AbstractFieldDescription field) {
78  		if (!isSameType(field.getMirror().getFieldTypeAsTypeElement(), knownType)) {
79  			return Collections.emptyList();
80  		}
81  		Collection<FieldDSLMethod> tmp = new ArrayList<>(Arrays.asList(builderFor(field, "after"),
82  				builderFor(field, "sameOrAfter"), builderFor(field, "before"), builderFor(field, "sameOrBefore")));
83  		if (withSameDay) {
84  			tmp.add(builderFor(field, "sameDay"));
85  		}
86  		return Collections.unmodifiableCollection(tmp);
87  	}
88  
89  	/*
90  	 * (non-Javadoc)
91  	 * 
92  	 * @see ch.powerunit.extensions.matchers.provideprocessor.extension.
93  	 * AutomatedExtension#accept(ch.powerunit.extensions.matchers.
94  	 * provideprocessor.ProvidesMatchersAnnotatedElementData)
95  	 */
96  	@Override
97  	public Collection<Supplier<DSLMethod>> accept(ProvidesMatchersAnnotatedElementData clazz) {
98  		return Collections.emptyList();
99  	}
100 
101 }
